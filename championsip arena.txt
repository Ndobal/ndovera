🔐 Middleware (Auth Guard)
export const authMiddleware = async (req, res, next) => {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) return res.status(401).json({ error: "Unauthorized" });

  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;
    next();
  } catch {
    return res.status(401).json({ error: "Invalid token" });
  }
};

import bcrypt from "bcrypt";
import jwt from "jsonwebtoken";
import prisma from "../../config/db.js";

export const register = async (req, res) => {
  const { email, password } = req.body;

  const hash = await bcrypt.hash(password, 10);

  const user = await prisma.user.create({
    data: {
      email,
      password: hash,
      role: "student",
      tenantId: "school_1"
    }
  });

  res.json(user);
};

export const login = async (req, res) => {
  const { email, password } = req.body;

  const user = await prisma.user.findUnique({ where: { email } });

  const valid = await bcrypt.compare(password, user.password);

  if (!valid) return res.status(401).json({ error: "Invalid" });

  const token = jwt.sign(
    { id: user.id, tenantId: user.tenantId },
    process.env.JWT_SECRET,
    { expiresIn: "1h" }
  );

  res.json({ token });
};

🧾 6. SUBMISSION SYSTEM
export const submitAnswer = async (req, res) => {
  const { questionId, answer, timeTaken } = req.body;

  const question = await prisma.question.findUnique({
    where: { id: questionId }
  });

  const isCorrect = question.correctAnswer === answer;

  await prisma.submission.create({
    data: {
      ...req.body,
      isCorrect
    }
  });

  if (isCorrect) {
    await prisma.participant.update({
      where: { id: req.body.participantId },
      data: { score: { increment: 1 } }
    });
  }

  res.json({ isCorrect });
};

export const getSpellingData = async (word) => {
  const cached = await redis.get(word);

  const data = await generateSpellingData(word);

  await redis.set(word, JSON.stringify(data), "EX", 86400);

  return data;
};

  const matches = [];

    matches.push({
      player1: players[i],
      player2: players[players.length - 1 - i]
    });
  }

  return matches;
};

const awardXP = async (userId, points) => {
  const user = await prisma.user.findUnique({ where: { id: userId } });
  const newXP = user.xp + points;

  // Level up every 100 XP
  const newLevel = Math.floor(newXP / 100) + 1;

  await prisma.user.update({
    where: { id: userId },
    data: { xp: newXP, level: newLevel }
  });
};
