// ================= UPGRADE / RESTRICTION LOGIC =================


export const canAddSchool = (owner) => {
  if (owner.tier === "enterprise") return true;
  return owner.currentSchools < owner.maxSchools;
};


export const handleAddSchool = (owner) => {
  if (!canAddSchool(owner)) {
    window.location.href = "/upgrade";
  } else {
    window.location.href = "/create-school";
  }
};




// ================= ADMIN PRICING CONTROL PANEL =================


export function AdminPricingPanel() {
  const [prices, setPrices] = useState({
    growth: 475,
    pro: 450,
  });


  const updatePricing = async () => {
    await fetch("/api/admin/pricing", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(prices),
    });


    alert("Pricing updated");
  };


  return (
    <div className="p-6">
      <h2 className="text-2xl font-bold mb-4">Pricing Control</h2>


      <input
        type="number"
        value={prices.growth}
        onChange={(e) => setPrices({ ...prices, growth: e.target.value })}
      />


      <input
        type="number"
        value={prices.pro}
        onChange={(e) => setPrices({ ...prices, pro: e.target.value })}
      />


      <button onClick={updatePricing} className="bg-black text-white p-2 mt-4">
        Save
      </button>
    </div>
